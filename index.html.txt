<script>
// ===== NAVIGATION FUNCTION =====
function showSection(id){
  const sections=['home','login','papers','discussions','dashboard'];
  sections.forEach(sec => {
    document.getElementById(sec).style.display = (sec===id)?'block':'none';
  });
}

// ===== ACCOUNTS =====
let accounts = [];
let currentUser = null;

function createAccount(){
  const name=document.getElementById('newName').value;
  const user=document.getElementById('newUser').value;
  const pass=document.getElementById('newPass').value;
  if(name && user && pass){
    accounts.push({name,user,pass,papers:[],average:0});
    alert('Account Created! You can login now.');
  } else alert('Fill all fields!');
}

function login(){
  const user=document.getElementById('loginUser').value;
  const pass=document.getElementById('loginPass').value;
  const acc=accounts.find(a=>a.user===user && a.pass===pass);
  if(acc){
    currentUser = acc;
    alert('Login Successful!');
    showSection('dashboard');
    updateDashboard();
  } else alert('Invalid login!');
}

// ===== DASHBOARD =====
let weeks=[];
let marks=[];
const ctx=document.getElementById("marksChart").getContext("2d");
let chart=new Chart(ctx,{
  type:'line',
  data:{labels:weeks,datasets:[{label:'Marks',data:marks, borderColor:'#ff7a00', tension:0.3, fill:false}]},
  options:{scales:{y:{beginAtZero:true, max:100}}}
});

function addMarks(){
  let week=document.getElementById("week").value;
  let mark=document.getElementById("marks").value;
  if(week && mark && currentUser){
    weeks.push("Week "+week);
    marks.push(parseInt(mark));
    chart.update();

    currentUser.papers.push(parseInt(mark));
    let total=currentUser.papers.reduce((a,b)=>a+b,0);
    currentUser.average=(total/currentUser.papers.length).toFixed(2);

    updateDashboard();
  } else alert("Enter week, marks, and login first!");
}

function updateDashboard(){
  if(currentUser){
    document.getElementById("studentName").innerText=currentUser.name;
    document.getElementById("papersWritten").innerText=currentUser.papers.length;
    document.getElementById("avgMark").innerText=currentUser.average;

    // Grade calculation
    let grade='N/A';
    let avg = parseFloat(currentUser.average);
    if(avg>=90) grade='A+';
    else if(avg>=80) grade='A';
    else if(avg>=70) grade='B+';
    else if(avg>=60) grade='B';
    else grade='C';
    document.getElementById("grade").innerText=grade;
  }
}
